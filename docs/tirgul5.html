קריאת המערכת open()

#include <sys/stat.h> #include <fcntl.h> int open(const char *pathname, int flags);

הפרמטר flags קובע את אופן הפתיחה:

דגלמשמעות
O_RDONLYקריאה בלבד
O_WRONLYכתיבה בלבד
O_RDWRקריאה וכתיבה
O_CREATיצירת הקובץ אם אינו קיים
O_EXCLגורם ל-open להיכשל אם הקובץ כבר קיים
O_TRUNCקיצוץ גודל הקובץ לאפס
O_APPENDהכתיבה תהיה לסוף הקובץ (לא דורסים)

הצלחה → fd, אחרת -1.

ניהול תיקיות

נעבור מרמת הקובץ הבודד לקריאת תוכן של תיקייה שלמה.

#include <dirent.h> DIR* opendir(const char* pathname); // מצביע ל-DIR או NULL struct dirent* readdir(DIR* dp); // רשומה אחת בכל קריאה int closedir(DIR* dp);

opendir מחזירה מצביע ל-stream של מידע, שבעת היצירה ממוקם ברשומה הראשונה בתיקייה. readdir פועלת כמו איטרטור — כל קריאה מחזירה את הרשומה הבאה, ו-NULL כשנגמרים הקבצים. struct dirent מכיל שני שדות עיקריים: d_fileno (מספר ה-i-node) ו-d_name (שם הקובץ).

scan_dir.c
#include <stdio.h>
#include <dirent.h>

int main(int argc, char **argv) {
    DIR *pDir;
    struct dirent *pDirent;

    if ((pDir = opendir(argv[1])) == NULL)
        exit(1);

    /* מעבר על כל רשומות התיקייה והדפסת שמותיהן */
    while ((pDirent = readdir(pDir)) != NULL)
        printf("%s\n", pDirent->d_name);

    closedir(pDir);
    return 0;
}

הקלט: argv[1] — נתיב לתיקייה. הפלט: שמות כל הקבצים והתיקיות שבתוכה, שורה לכל אחד.

read · write · close

int read(int fd, void* buff, size_t nbytes); // שואב נתונים מקובץ לזיכרון int write(int fd, void* buff, size_t nbytes); // כותב מהזיכרון לקובץ int close(int fd);
  • read — buff הוא היעד בזיכרון. הקריאה מתבצעת עד שהגענו ל-nbytes או עד סוף הקובץ.
  • write — מעביר מ-buff לקובץ. הצלחה → מספר הבתים שנכתבו, כישלון → −1.
  • close — מעדכן את ה-open file description שיש לו מצביע אחד פחות. כשהמונה מגיע לאפס, ה-OFD משתחרר.

דוגמה: העתקת קובץ

copy_file.c
#include <fcntl.h>
#include <unistd.h>
#define SIZE 4096

int main(void) {
    int fdin, fdout;
    char buf[SIZE + 1];
    int charsr;   /* chars read */

    fdin = open("dugma1.txt", O_RDONLY);
    if (fdin < 0) { perror("after open"); _exit(-1); }

    /* יצירה לקריאה-כתיבה עם הרשאות 0666 */
    fdout = open("dugma2.txt", O_CREAT | O_RDWR, 0666);
    if (fdout < 0) { perror("after create"); _exit(-1); }

    do {
        charsr = read(fdin, buf, SIZE);
        write(fdout, buf, charsr);   /* כותבים בדיוק מה שנקרא */
    } while (charsr == SIZE);

    close(fdin);
    close(fdout);
    return 0;
}

הסבר משולב: פותחים מקור לקריאה ויעד ליצירה+כתיבה. בכל איטרציה read ממלא את buf וה-קלט ל-write הוא בדיוק charsr (מספר הבתים שנקראו) — לא SIZE, אחרת היינו כותבים זבל בסוף. הלולאה נעצרת כשקראנו פחות מ-SIZE (סוף קובץ).

משפחת dup

#include <unistd.h> int dup(int oldfd); // עותק באינדקס הפנוי הנמוך ביותר int dup2(int oldfd, int newfd); // מאלץ את newfd להצביע לאן ש-oldfd מצביע

הקריאות משכפלות fd — כמה אינדקסים בטבלה מצביעים על אותו קובץ פתוח. למה? עבור ניתוב מחדש. אם כתבנו תוכנית עם הרבה הדפסות, במקום לשנות את כל הקוד — נשתמש ב-dup2 כדי לומר למערכת "מעכשיו כל מה שאמור ללכת למסך, שלחי אותו ל-fd של הקובץ".

I/O Redirection

היכולת להפנות קלט/פלט סטנדרטי למקומות אחרים. ls > file.txt ינתב את ההדפסה לקובץ. מאחורי הקלעים זה ממומש ב-dup: פותחים את הקובץ ומקבלים fd, ואז dup2 מעתיק את ה-fd של הקובץ לתוך ה-fd של ה-IO.

redirect.c
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

int main(int argc, char **argv) {
    printf("This goes to the standard output.\n");

    int newfd = open(argv[1], O_CREAT | O_WRONLY | O_TRUNC, 0644);
    printf("Now the standard output will go to \"%s\".\n", argv[1]);

    dup2(newfd, 1);   /* כל מה שנכתב ל-stdout (fd=1) ילך לקובץ */

    printf("This goes to the file.\n");  /* כבר לא למסך! */
    return 0;
}

הסבר: אחרי dup2(newfd, 1), ה-fd מספר 1 (stdout) מצביע על הקובץ. לכן ההדפסה האחרונה נכתבת לקובץ ולא למסך.

lseek — גישה לכל מקום בקובץ

#include <sys/types.h> #include <unistd.h> off_t lseek(int fd, off_t offset, int whence);

משנה את ה-offset הנוכחי של ה-fd. whence:

  • SEEK_SET — ההיסט מתחילת הקובץ.
  • SEEK_CUR — ההיסט מהמיקום הנוכחי.
  • SEEK_END — ההיסט מסוף הקובץ.

הצלחה → מיקום הסמן החדש, אחרת −1.

file_size.c
/* טריק לקבלת גודל קובץ: קופצים לסוף ומקבלים את ההיסט */
off_t size = lseek(fd, 0, SEEK_END);

chmod

#include <sys/stat.h> int chmod(const char* path, mode_t mod);

אחראית על אבטחה והרשאות במערכת הקבצים — משנה מי רשאי לקרוא/לכתוב/להריץ קובץ. mod מיוצג לרוב כערך אוקטלי או כשילוב ביטים.

errno · perror

#include <errno.h> extern int errno; #include <stdio.h> void perror(const char *s);

כמעט כל קריאת מערכת מעדכנת את המשתנה הגלובלי errno בקוד מספרי שמייצג את סיבת השגיאה. perror מדפיסה ל-stderr את ה-string שהועבר, ואז נקודתיים, רווח, והודעת השגיאה המתאימה ל-errno.

האם כל read מבצעת קריאה פיזית מהדיסק?

לא! קריאה מהדיסק יקרה מאוד. לכן נוקטים באסטרטגיית הבלוקים: כשהתוכנית מבקשת בייט בודד, המערכת קוראת בלוק שלם ושומרת אותו בקאש. בקשה לבייט הבא תגיע ישר מהזיכרון המהיר.

📌 fread / fwrite מול read / write

פונקציות ספריית f מנהלות חוצץ פנימי ב-user-space. רק כשהבאפר מתמלא — מתבצעת קריאת מערכת אחת גדולה. זה חוסך overhead של מעבר user↔kernel, אבל המחיר: אם התהליך קורס, מידע שנשאר בבאפר יעלם.

כדי לחייב כתיבה פיזית — 2 דרכים:

  • דגל O_SYNC בפתיחת קובץ — כל כתיבה חסומה עד שהמערכת מאשרת שהמידע נכתב פיזית לדיסק.
  • הפונקציה fsync — מכריחה לרוקן את כל הבאפר לדיסק בנקודת זמן מסוימת (אחרי סדרת כתיבות).

בנוסף: fprintf מאפשרת הדפסה בפורמט printf לתוך קובץ, ו-fscanf קריאה בפורמט scanf מקובץ.

🎁 סיפור סוף התרגול

📖 תרגול 5 — "הכל זה קובץ"

הפילוסופיה של Unix אומרת ש-file descriptors הם לא רק לקריאת טקסט. ב-Linux: /dev/null זורק הכל, /dev/random מייצר בתים אקראיים, ו-/proc/cpuinfo מראה מידע על ה-CPU — אבל הוא נוצר בזמן אמת ע"י הקרנל; אין קובץ אמיתי בדיסק.

הקריאה dup() שלמדתם היא האופן שבו ה-shell מממש I/O redirection: כשאתם כותבים ls > output.txt, ה-shell קורא ל-dup2() כדי להחליף את ה-fd של stdout באחד שמצביע לקובץ.

המסר: הכוח האמיתי לא במורכבות הכלים, אלא בפשטות הממשק. כשמזקקים את המציאות לשפה אחת אחידה — מפסיקים לנהל בעיות נקודתיות ומתחילים לשלוט בארכיטקטורה של הפתרון.
