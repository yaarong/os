דרכי שליחת סיגנלים

  • ישירות מהמקלדת — Ctrl+C שולח SIGINT (עצירה מיידית), Ctrl+Z שולח SIGTSTP (השהיה, לא הריגה).
  • מהטרמינל — kill -signalNum/signalName pid שולח סיגנל לכל תהליך.
  • באמצעות system calls — שליחה מקוד תכנות.

בעיית סנכרון

כשתהליך מקבל סיגנל, הקרנל עוצר את זרם הריצה ומפנה את ה-CPU לפונקציית הטיפול. אבל מה קורה אם בזמן הטיפול בסיגנל א' מגיע סיגנל ב'? מצב כזה עלול להוביל לבעיות סנכרון קשות. הפתרון: חסימת סיגנלים.

  • השהיה זמנית — סיגנל חסום שנשלח עובר למצב PENDING: המערכת רושמת שהגיע, אך לא מאפשרת לו לקטוע את הריצה.
  • כשהקוד הרגיש מסתיים והחסימה מוסרת, הקרנל בודק את התור ומספק את הסיגנלים מיד.
  • עיקרון חשוב: אם בזמן חסימה נשלחו 5 סיגנלים של SIGINT — המערכת לא תשמור את כל החמישה. פונקציית הטיפול תתבצע פעם אחת בלבד.
  • שני סיגנלים שלא ניתן לחסום — SIGKILL, SIGSTOP.

sigprocmask

#include <signal.h> int sigprocmask(int how, const sigset_t *set, sigset_t *oldset);

מאפשרת לתהליך לשנות את ה-signal mask שלו. שלושה פרמטרים:

  1. how — כיצד לשנות את המסכה. למשל SIG_BLOCK — הסיגנלים שב-set יתווספו למסכת החסימה הקיימת.
  2. set — מצביע לקבוצת הסיגנלים שעליהם נחיל את הפעולה.
  3. oldset — מצביע למבנה שבו תישמר מסכת החסימה הישנה (לשחזור מאוחר). ניתן לשלוח NULL.

ערך חזרה: 0 בהצלחה, −1 בשגיאה.

פעולות על קבוצות סיגנלים (sigset_t)

int sigemptyset(sigset_t *set); // קבוצה ריקה int sigfillset(sigset_t *set); // קבוצה מלאה (כל הסיגנלים) int sigaddset(sigset_t *set, int signum); // הוספת סיגנל int sigdelset(sigset_t *set, int signum); // הסרת סיגנל int sigismember(sigset_t *set, int signum); // בדיקת חברות

sigemptyset/sigfillset/sigaddset/sigdelset מחזירים 0 בהצלחה, −1 בכישלון. sigismember מחזיר 1 אם signum בקבוצה, 0 אם לא, −1 בכישלון.

דוגמה: חסימת SIGUSR1

block_demo.c
#include <signal.h>

int main(void) {
    /* נניח שכבר חסומים SIGSEGV ו-SIGTERM */
    sigset_t x, y;

    sigemptyset(&x);            /* מנקה ערכי זבל מהקבוצה */
    sigaddset(&x, SIGUSR1);     /* מוסיף SIGUSR1 לקבוצה */

    /* מוסיף את x למסכה הקיימת, שומר את הישנה ב-y */
    sigprocmask(SIG_BLOCK, &x, &y);

    /* ... קטע קריטי שבו SIGUSR1 לא יפריע ... */

    sigprocmask(SIG_SETMASK, &y, NULL);   /* שחזור המצב הקודם */
    return 0;
}

הסבר משולב: מאתחלים קבוצה ריקה x, מוסיפים אליה את SIGUSR1, ואז sigprocmask(SIG_BLOCK, &x, &y) מוסיף אותו לחסימה הקיימת ושומר את המסכה הישנה ב-y. בסיום הקטע הקריטי משחזרים בדיוק את המצב הקודם.

⚠️ למה sigprocmask לא מספיק?

הבעיה בנקודת הזמן: פעולת החסימה דורשת לכתוב שורות קוד ולקרוא לפונקציה — וכל עוד לא הגענו אליהן, התוכנית חשופה. למשל: ייתכן שכבר נכנסנו לפונקציית הטיפול והגיע סיגנל נוסף לפני שביצענו חסימה. לכן נדרש מנגנון מובנה יותר — sigaction.

sigaction

#include <signal.h> int sigaction(int signum, const struct sigaction *act, struct sigaction *oldact);

בניגוד ל-signal(), ה-sigaction() דואגת לחסימה של סיגנלים מסוימים במהלך הטיפול. הצלחה → 0, כישלון → −1. act מצביע להגדרות החדשות, oldact ישמור את הקודמות.

מבנה struct sigaction:

  • sa_handler — מצביע לפונקציית הטיפול (מקבלת את מספר האיתות).
  • sa_mask — הפתרון לבעיית הסנכרון! מגדיר אילו איתותים ייחסמו אוטומטית בזמן שה-handler רץ. בנוסף, המערכת חוסמת אוטומטית גם את האיתות שגרם לטיפול — מונע ריצה רקורסיבית של ה-handler על עצמו.
  • sa_flags — אוסף דגלים (מופרדים ב-OR):
    • SA_NODEFER — הסיגנל הנוכחי לא ב-mask (בעייתי לסנכרון).
    • SA_SIGINFO — להשתמש ב-sa_sigaction במקום ב-sa_handler.
    • SA_RESETHAND — מחזיר את ה-handler ל-SIG_DFL בכניסה אליו.
sigaction_demo.c
#include <stdio.h>
#include <signal.h>
#include <unistd.h>

void handler(int sig) {
    printf("handling %d safely...\n", sig);
    sleep(2);   /* גם אם יגיעו SIGINT נוספים — הם ייחסמו עד שנסיים */
}

int main(void) {
    struct sigaction act;
    act.sa_handler = handler;
    sigemptyset(&act.sa_mask);
    sigaddset(&act.sa_mask, SIGINT);   /* חסום SIGINT בזמן הטיפול */
    act.sa_flags = 0;

    sigaction(SIGINT, &act, NULL);

    while (1) pause();
    return 0;
}

הסבר משולב: ממלאים struct sigaction: sa_handler מצביע לפונקציה, וב-sa_mask מוסיפים את SIGINT כך שגם אם יגיעו SIGINT נוספים בזמן ה-handler — הם ייחסמו ויטופלו אחר כך (פעם אחת). זהו היתרון על signal() — אטומיות וחסימה מובנית.

בדיקת איתותים ממתינים — sigpending

int sigpending(sigset_t *set);

מקבלת set ריק וממלאת אותו באיתותים שממתינים (במצב PENDING). מחזירה 0 בהצלחה, −1 בכישלון. כדי לבדוק אם איתות מסוים קיים ב-set — משתמשים ב-sigismember.

ירושת סיגנלים

שאלהתשובה
האם תהליך בן יורש את הסיגנלים הממתינים לאב?לא
האם תהליך בן יורש את signal handlers ואת ה-signal mask מהאב?כן

כדי להתעלם מהסיגנלים אחרי הורדת החסימה, ניתן להגדיר להם את פעולת SIG_IGN.

🎁 סיפור סוף התרגול

📖 תרגול 8 — חולשת OpenSSH (2006)

בקוד של שרתי OpenSSH הוגדר Signal Handler לטיפול ב-SIGALRM (שמופעל אם המשתמש לא מסיים להתחבר בזמן קצוב). בגלל אי-חסימה של איתותים, ה-SIGALRM הגיע בדיוק כשהתוכנית הראשית הייתה באמצע malloc() או free().

מטפל האיתותים ניסה בעצמו לבצע פעולת ניהול זיכרון, מה שגרם לשיבוש (Double Free). האקרים ניצלו את 'מרוץ האיתותים' (Race Condition) כדי לעקוף את מנגנון ההזדהות ולהשתלט על השרתים מרחוק.

המסר: בדיוק כמו ש-sigprocmask חוסם רעשי רקע ומגן על קטעי קוד קריטיים — גם בחיים צריך לדעת מתי להפעיל 'מסיכת חסימה' על הסחות דעת כדי להתמקד במה שחשוב.
