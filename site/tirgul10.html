תקשורת בין תהליכים — IPC · PIPE · FIFO

גישות לתקשורת בין תהליכים, צינורות (pipe) חד-כיווניים לפי FIFO, קריאה וכתיבה חוסמות, SIGPIPE ואטומיות, וצינורות בעלי שם (FIFO) לתהליכים לא קרובים.

תקשורת בין תהליכים (IPC)

איך מעבירים מידע בין תהליכים? שתי גישות ראשוניות:

  • סיגנלים — בכל פעם שצד אחד רוצה להעביר מידע, הוא שולח סיגנל. אך כשרוצים להעביר נתונים — זה לא כלי מתאים.
  • קבצים — תהליך א' כותב לקובץ ותהליך ב' קורא ממנו. הקשיים: לא תוכנן כערוץ תקשורת ישיר, גישה לכונן איטית (Overhead), ובעיית סנכרון — תהליך הקריאה חייב לדעת בוודאות שהכתיבה הסתיימה, אחרת יקבל מידע לא שלם.

בתוך ה-Open File Description יש את ה-current file offset. כששני תהליכים משתמשים בקובץ כערוץ — הם מצביעים לאותו OFD וחולקים את אותו offset, ושניהם חייבים לדעת היכן הסמן בכל רגע.

קריאת המערכת lseek

#include <sys/types.h> #include <unistd.h> off_t lseek(int fd, off_t offset, int whence);

משנה את ה-current file offset, וכך שולטת מהיכן יתבצעו קריאה/כתיבה. whence: SEEK_SET (מתחילת הקובץ), SEEK_CUR (מהמיקום הנוכחי), SEEK_END (מסוף הקובץ).

PIPE

נגדיר PIPE כזרם נתונים חד-כיווני שמתנהל לפי FIFO (ראשון נכנס, ראשון יוצא). זהו פתרון IPC טוב יותר כי הוא מובנה לסנכרון תהליכים, ללא ניהול ידני של אופסטים.

  • הנתונים זורמים בכיוון אחד בלבד.
  • הנתונים יוצאים באותו סדר שנכנסו.
  • הגישה דרך שני fd נפרדים — אחד לקריאה בלבד והשני לכתיבה בלבד.
  • אין גישה אקראית — לא ניתן לומר "לך לבית 100"; הנתונים זזים, לא יושבים בכתובות.

ניתן לדמיין תהליך אחד שדוחף "כדורים" (יחידות מידע) לצינור, ותהליך שני בצד השני שתופס אותם. כשמערכת ההפעלה יוצרת את הצינור, היא נותנת זוג מפתחות — אחד לכתיבה ואחד לקריאה.

קריאת המערכת pipe

#include <unistd.h> int pipe(int filedes[2]);

שולחים מערך של 2 int, והפונקציה ממלאת אותו ב-FD: filedes[0] לקריאה, filedes[1] לכתיבה. ערך מוחזר: 0 בהצלחה, −1 בכישלון.

📌 צינור פרטי + fork

הצינור פרטי לתהליך שיצר אותו — תהליך זר לא יודע שהוא קיים. בעת fork(), הבן מקבל עותק של טבלת ה-FD של האב, ולכן יש לו אותו FD שמצביע לאותו צינור. לאחר ה-fork יש לסגור את הקצוות המיותרים: אם האב כותב הוא לא צריך את fd הקריאה, ואם הבן קורא הוא לא צריך את fd הכתיבה — אחרת התוכנית עלולה להיתקע.

pipe_demo.c
#include <stdio.h>
#include <unistd.h>
#include <string.h>

int main(void) {
    int fd[2];
    pipe(fd);                 /* fd[0]=קריאה , fd[1]=כתיבה */

    if (fork() == 0) {
        /* --- הבן: קורא --- */
        close(fd[1]);                 /* לא צריך את קצה הכתיבה */
        char buf[64];
        int n = read(fd[0], buf, sizeof buf);
        buf[n] = '\0';
        printf("Child received: %s\n", buf);
        close(fd[0]);
    } else {
        /* --- האב: כותב --- */
        close(fd[0]);                 /* לא צריך את קצה הקריאה */
        char *msg = "Hello from parent";
        write(fd[1], msg, strlen(msg));
        close(fd[1]);                 /* סגירה → שולח EOF לקורא */
    }
    return 0;
}

הסבר משולב: יוצרים צינור עם pipe(fd), ואז fork. כל צד סוגר את הקצה שאינו צריך: האב סוגר את הקריאה והבן את הכתיבה. האב כותב קלט = המחרוזת "Hello from parent" לתוך fd[1], והבן קורא אותה מ-fd[0]. סגירת קצה הכתיבה ע"י האב היא שמסמנת לבן את סוף הנתונים (EOF).

קריאה מ-PIPE

read example
read(filedes[0], buff, BUFF_SIZE);
  • בניגוד לקובץ רגיל (שמחזיר 0 בתים אם ריק) — קריאה מצינור ריק כשעדיין יש תהליכים שמחזיקים את ה-write descriptor פתוח → התהליך נחסם ומחכה שייכתבו נתונים חדשים.
  • כשהתהליך מתעורר, read מחזירה את כמות הנתונים הזמינה. היא מחזירה 0 אם ורק אם כל ה-write descriptors נסגרו — סימן לקורא שאין מה לחכות יותר.
  • בזמן ההמתנה התהליך ישן ולא מבזבז CPU.

כתיבה ל-PIPE

יש מקום פנוי

אם יש תהליך קורא פעיל — הכתיבה מתבצעת במלואה. אם כל ה-read descriptors נסגרו — המערכת שולחת SIGPIPE ומחזירה כישלון (אין למי לכתוב).

הצינור מלא

אם יש קורא פעיל — הכתיבה נחסמת וממתינה עד שיתפנה מקום. אם אין קורא — SIGPIPE וכישלון.

  • כתיבה של כמות קטנה מ-BUF_SIZE מובטחת להיות אטומית — אם שני תהליכים כותבים בו-זמנית, הנתונים לא יתערבבו.
  • גודל ה-PIPE מוגבל.

FIFO — צינור בעל שם

הבעיה ב-PIPE: רק תהליכים קרובי משפחה (אב ובן) יכולים לתקשר. הפתרון: לתת לצינור שם כמו לקובץ. כל תהליך שיודע את הנתיב יכול לפתוח אותו עם open() רגיל, ומערכת ההפעלה מקשרת בין הצדדים בדיוק כמו ב-PIPE רגיל.

  • PIPE ציבורי שדרכו כל התהליכים יכולים לתקשר.
  • נוצר ע"י mknod() או mkfifo().
  • שם ה-FIFO הוא כשם קובץ במערכת הקבצים, למרות שאינו קובץ כלל.
  • פתיחה לקריאה בלבד נחסמת עד שתהליך אחר יפתח לכתיבה (ולהפך). פתיחה לקריאה+כתיבה יחד — לא חוסמת.
  • לא משוחרר אוטומטית — יש לפנותו במפורש (rm).
#include <sys/types.h> #include <sys/stat.h> int mknod(const char *pathname, mode_t mode, dev_t dev); int mkfifo(const char *pathname, mode_t mode);
fifo_demo.c
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>

#define MODE_MKNOD  (S_IFIFO | S_IRUSR | S_IWUSR)
#define MODE_MKFIFO (S_IRUSR | S_IWUSR)

int main(void) {
    /* יצירת ה-FIFO בשם myfifo.f */
    mknod("myfifo.f", MODE_MKNOD, 0);
    /* או, חלופה פשוטה יותר: */
    mkfifo("myfifo.f", MODE_MKFIFO);

    /* כעת פותחים אותו כמו קובץ רגיל */
    int fd = open("myfifo.f", O_RDONLY);
    /* ... read/write דרך fd ... */
    return 0;
}

הסבר משולב: mknod/mkfifo יוצרים קובץ מסוג FIFO בשם myfifo.f עם הרשאות מתאימות. מרגע שהוא קיים בשם, כל תהליך — גם לא קרוב משפחה — יכול לפתוח אותו עם open(); אחד יפתח לקריאה והשני לכתיבה, והנתונים יזרמו בזיכרון הקרנל.

📖 תרגול 10 — לידת ה-Pipe

ב-1973, דאג מקאילרוי מ-Bell Labs הציע את ה-pipe — דרך לחבר את הפלט של תוכנה אחת לקלט של אחרת. קן תומפסון אהב את הרעיון כל כך שמימש pipes בקרנל Unix בלילה אחד. בבוקר שאחרי, כולם ב-Bell Labs שרשרו תוכנות קטנות יחד: cat file | grep pattern | sort | uniq -c.

הפילוסופיה הזו — תוכנות קטנות שעושות דבר אחד טוב, מחוברות דרך pipes — קדמה לארכיטקטורת microservices ב-40 שנה. הרעיון זהה: רכיבים עצמאיים שמתקשרים דרך ממשקים מוגדרים היטב.

המסר: בחיים תצטרכו Pipes — שיתופי פעולה פשוטים שיוצרים שלם הגדול מסך חלקיו; אך כדי לחבר בין עולמות שונים לחלוטין, ייתכן ותצטרכו FIFOs — ערוצי תקשורת קבועים, עצמאיים ובעלי שם.
