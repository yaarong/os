יצירת תהליכים — fork · exec · exit

מזהי תהליך

PID

מספר ייחודי שמתקבל בזמן יצירת התהליך.

UID

מספר ייחודי של המשתמש שמריץ את התהליך.

PPID

המזהה של התהליך שיצר את התהליך הנוכחי (Parent).

Mode / Priority

מצב התהליך (נוצר, ממתין, מחכה...) ועדיפות ההרצה שלו.

כאשר מערכת ההפעלה נטענת נוצרים תמיד 2 תהליכים: swapper (Pid=0, ניהול הזיכרון) ו-init (Pid=1, כל התהליכים הופכים לצאצאים שלו).

fork()

הפעולה משכפלת את תהליך האב לתהליך בן, וממשיכה לשורת הקוד הבאה בשני התהליכים. לשני התהליכים:

  • קוד זהה (מיד אחרי הקריאה ל-fork)
  • זיכרון זהה (משתנים וערכיהם) — אך בעותקים נפרדים!
  • סביבה זהה (קבצים פתוחים, fd וכו')
  • PID שונה!
pid_t fork(void); // <sys/types.h> , <unistd.h>
📌 ערך מוחזר

כישלון → -1 לאב. הצלחה → לבן מוחזר 0, ולאב מוחזר ה-PID של הבן (גדול מ-0). זו הדרך להבדיל ביניהם בקוד.

fork_basic.c
#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>

int main(void) {
    pid_t pid;
    if ((pid = fork()) == 0)
        printf("1");      /* תהליך הבן (fork החזיר 0) */
    else
        printf("2");      /* תהליך האב (fork החזיר pid > 0) */
    printf("3");          /* שני התהליכים מבצעים זאת */
    return 0;
}

מה יודפס? לא ניתן לדעת! (לא דטרמיניסטי, תלוי מערכת הפעלה). הדבר היחיד שבטוח: 3 לא יודפס בתחילה. אפשרויות: 1323, 2313, 2133, 1233.

מנגנון Copy-on-write

ייחודי ללינוקס: לאחר fork, שני התהליכים משתמשים בדפים משותפים עד לרגע שבו אחד מהם מבקש לשנות דף. אז מערכת ההפעלה מעתיקה את הדף — כך שלכל תהליך יהיה עותק פרטי — ורק אז מאפשרת את הכתיבה. כלומר, ההעתקה בפועל קורית רק כשיש שינוי.

ספירת תהליכים

many_forks.c
fork();
fork();
/* ... */
fork();   /* נקרא n פעמים ברצף */

כל תהליך פותח 2 חדשים — כמו עץ בינארי מלא. מספר התהליכים = מספר העלים = 2n. מספר התהליכים החדשים שנוצרו = 2n−1.

⚠️ יתום (orphan)

כאשר אב מסיים והבן עדיין חי — הבן נחשב יתום. התהליך init (pid=1) מאמץ דיפולטיבית את כל היתומים (עבורם PPID=1).

פונקציות שימושיות

pid_t getpid(void); // מזהה התהליך הקורא pid_t getppid(void); // מזהה תהליך האב uid_t getuid(void); // מזהה המשתמש של התהליך הקורא

דוגמה: אינסוף תהליכים

infinite.c
pid_t pid;
pid = getpid();
while (fork() == 0) {
    if (pid == getpid())
        break;
}

פתרון: ייווצרו אינסוף תהליכים. התהליך המקורי קורא ל-fork, יוצר בן ומסיים (הערך שחזר אצלו ≠ 0). הבן נכנס ללולאה, ואצלו לא מתקיים pid == getpid() (כי ה-pid שונה משל האב), לכן הוא יוצר נכד ומסיים. בכל פעם "נוצר" ילד חדש שלא עומד בתנאי הברייק — וכך נוצר עץ בינארי בצורת "שרוך" אינסופי של היררכיית ילד-אב.

exec()

נרצה להריץ תוכנית חדשה מתוך תהליך. exec טוענת תוכנית חדשה לביצוע במקום התוכנית הנוכחית. הפרמטר הראשון הוא שם קובץ התוכנית, ומשם — מצביעים למחרוזות הפרמטרים. הפרמטר הראשון חייב להיות שם הקובץ (מועבר פעמיים), והאחרון חייב להיות NULL.

📌 ערך מוחזר

כישלון → -1. בהצלחה — הקריאה כלל לא חוזרת, כי exec החליפה את התוכנית בחדשה.

וריאנטים — exec{l,v}{e,p}:

  • l — הארגומנטים כרשימה מופרדת בפסיקים, מסתיימת ב-NULL.
  • v — הארגומנטים כמערך מצביעים *argv, האיבר האחרון NULL.
  • e — מאפשר להעביר במפורש מערך משתני סביבה לתהליך החדש.
  • p — משתמש ב-PATH לאיתור קובץ ההרצה (ללא צורך בנתיב מלא).
exec_variants.c
#include <unistd.h>

/* l + נתיב מלא */
execl("/bin/ls", "ls", "-r", "-t", "-l", NULL);

/* v + נתיב מלא */
char *args[] = { "/bin/ls", "-r", "-t", "-l", NULL };
execv("/bin/ls", args);

/* v + p : מחפש ב-PATH, אין צורך בנתיב מלא */
char *args2[] = { "ls", "-r", "-t", "-l", NULL };
execvp("ls", args2);

הקומבו הקלאסי: fork + exec + wait

fork_exec_wait.c
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main(void) {
    pid_t pid = fork();

    if (pid == 0) {
        /* הבן: מחליף את עצמו בפקודה ls -l */
        char *args[] = { "ls", "-l", NULL };
        execvp("ls", args);
        perror("execvp failed");   /* מגיעים לכאן רק אם exec נכשלה */
        _exit(1);
    } else {
        /* האב: ממתין לסיום הבן */
        int status;
        wait(&status);
        printf("Child finished. status=%d\n", WEXITSTATUS(status));
    }
    return 0;
}

הסבר משולב: האב יוצר בן עם fork. הבן מריץ execvp("ls", args) — הקלט הוא מערך הארגומנטים {"ls","-l",NULL}, וה-PATH ימצא את /bin/ls. מרגע שה-exec מצליחה, קוד הבן מוחלף לחלוטין ב-ls. האב קורא ל-wait וחוסם עד שהבן מסתיים, ואז מחלץ את קוד היציאה מ-status.

wait · waitpid

כשהילד רץ, ההורה ממשיך בפעולותיו וגם ממתין לסיום הילד.

#include <sys/types.h> #include <sys/wait.h> pid_t wait(int *status);
  • התהליך הקורא (האב) ימתין עד שאחד מבניו יסתיים.
  • הערך המוחזר — המזהה של הבן שהתהליך ממתין לו.
  • status — מכיל מידע על סטטוס הבן שהסתיים.

waitpid משהה את התהליך עד שתהליך בן ספציפי משנה את מצבו — שליטה גמישה דרך options:

pid_t waitpid(pid_t pid, int *status, int options);

Context switch

נניח ש-init (PID=1) יוצר תהליך נוסף (PID=2) שמתחרה בו על משאבי המערכת. מערכת ההפעלה תחליף בין PID=1 לכל נתוניו לבין PID=2, תוך שמירה על חוצץ בין התהליכים. לפני שתהליך "מורדם" — נתוניו נשמרים, ולפני ש"מתעורר" — נטענים אותם נתונים לאותם מקומות בדיוק. אם משאב נחוץ תפוס ע"י תהליך אחר — התהליך יורדם שוב עד התור הבא.

exit · atexit · זומבים

exit מסיימת את התהליך באופן מיידי. Flush הוא העברת נתונים מזיכרון זמני אל היעד הסופי (כמו דיסק).

void _exit(int status); // סיום מיידי, ללא flush void exit(int status); // סיום כמעט מיידי, כן מבצע flush int atexit(void (*function)(void)); // רישום פונקציה לסיום

atexit "רושמת" פונקציה שתיקרא בסיום נורמלי של תהליך (או דרך return מ-main). הפונקציות נקראות בסדר הפוך לרישום — כמו מחסנית. מחזירה 0 בהצלחה.

atexit_demo.c
#include <stdio.h>
#include <stdlib.h>

void bye_a(void) { printf("A\n"); }
void bye_b(void) { printf("B\n"); }

int main(void) {
    atexit(bye_a);    /* נרשמת ראשונה */
    atexit(bye_b);    /* נרשמת שנייה  */
    printf("main done\n");
    return 0;
}
/* פלט (סדר הפוך — מחסנית):
   main done
   B
   A                                  */
🧟 זומבים

כאשר בן מסתיים והאב לא המתין לו — הבן נחשב זומבי. ה-PCB שלו עדיין נשמר (כי האב עוד יכול לעשות wait), ורק אז ה-PCB יימחק. אם האב לא יעשה wait, הבן יהפוך ליתום ו-init יאמץ אותו ויעשה לו wait.

📖 תרגול 3 — Fork Bomb

ה-fork bomb הקלאסי :(){ :|:& };: הוא רק 13 תווים — והוא יכול להפיל כמעט כל מערכת Linux לא מוגנת תוך שניות. זוהי פונקציה שקוראת לעצמה פעמיים, ויוצרת צמיחה מעריכית של תהליכים. fork bomb בודד יכול ליצור מעל מיליון תהליכים בפחות מ-60 שניות.

בתקרית אמיתית, pull request זדוני ב-GitHub Actions הכיל fork bomb בתוך סוויטת הבדיקות, מה שהפיל את כל ה-runner host והשפיע על builds אחרים.

המסר: גם פקודה תמימה למראה יכולה להוריד שרתים שלמים. מי שמנסה לבלוע יותר מדי תהליכים "במזלג" אחד — עלול להחניק את כל המערכת. תפסת מרובה — לא תפסת.
